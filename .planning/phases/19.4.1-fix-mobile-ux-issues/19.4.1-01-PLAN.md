**1. Refactor mobile/constants/config.ts:**

Change `API_KEY` from `const` to `let` to enable ES module live binding. Add `setRuntimeApiKey()` function. This allows the provider to update the export value at runtime, and ALL existing import sites automatically see the new value without modification.

```typescript
export const API_BASE_URL = process.env.EXPO_PUBLIC_API_URL ?? "http://localhost:8003";
// Live binding: changes via setRuntimeApiKey() propagate to all importers
export let API_KEY = process.env.EXPO_PUBLIC_API_KEY ?? "";
export const USER_ID = "will";
export const MAX_FOLLOW_UPS = 2;

/**
 * Update the API key at runtime (called by ApiKeyProvider after SecureStore read).
 * Uses ES module live binding -- all modules importing API_KEY see the new value.
 */
export function setRuntimeApiKey(key: string) {
  API_KEY = key;
}
```

**2. Create mobile/contexts/ApiKeyContext.tsx:**

React context that reads from SecureStore on mount, provides `{ apiKey, setApiKey, isLoading }` to the component tree. In dev builds where `EXPO_PUBLIC_API_KEY` env var is set, skips SecureStore and uses the env var directly (D-12). On mount AND on setApiKey, calls `setRuntimeApiKey()` from config.ts so utility modules that import `API_KEY` directly also get the updated value.

```typescript
import * as SecureStore from "expo-secure-store";
import { createContext, useContext, useState, useEffect, ReactNode } from "react";
import { setRuntimeApiKey } from "../constants/config";

interface ApiKeyContextType {
  apiKey: string;
  setApiKey: (key: string) => Promise<void>;
  isLoading: boolean;
}

const ApiKeyContext = createContext<ApiKeyContextType>({
  apiKey: "",
  setApiKey: async () => {},
  isLoading: true,
});

export function ApiKeyProvider({ children }: { children: ReactNode }) {
  const [apiKey, setApiKeyState] = useState("");
  const [isLoading, setIsLoading] = useState(true);

  useEffect(() => {
    const envKey = process.env.EXPO_PUBLIC_API_KEY;
    if (envKey) {
      setApiKeyState(envKey);
      setRuntimeApiKey(envKey);
      setIsLoading(false);
      return;
    }
    SecureStore.getItemAsync("api_key").then((stored) => {
      const key = stored ?? "";
      setApiKeyState(key);
      setRuntimeApiKey(key);
      setIsLoading(false);
    });
  }, []);

  const setApiKey = async (key: string) => {
    await SecureStore.setItemAsync("api_key", key);
    setApiKeyState(key);
    setRuntimeApiKey(key);
  };

  return (
    <ApiKeyContext.Provider value={{ apiKey, setApiKey, isLoading }}>
      {children}
    </ApiKeyContext.Provider>
  );
}

export const useApiKey = () => useContext(ApiKeyContext);
```
