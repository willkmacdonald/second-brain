```python
def trace_headers(request_id: str | None) -> dict[str, Any]:
    """Return kwargs dict with initial_headers for Cosmos calls.
    Returns empty dict when request_id is falsy."""
    if not request_id:
        return {}
    return {"initial_headers": {"x-ms-client-request-id": request_id}}
```
