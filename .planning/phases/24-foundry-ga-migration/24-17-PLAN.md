    Create `backend/scripts/backfill_inbox_session_id.py` with this body:

    ```
    """One-shot backfill: rename Inbox.foundryThreadId field to sessionId in Cosmos.

    Run from laptop BEFORE the Phase 24 deploy push (as part of pre-deploy gates).
    Idempotent — safe to re-run.

    Usage:
        cd backend && uv run python -m scripts.backfill_inbox_session_id

    Per Phase 24 task group 23.3 plan 24-17.
    """

    from __future__ import annotations

    import asyncio
    import logging
    import os
    import sys

    from azure.cosmos.aio import CosmosClient
    from azure.identity.aio import DefaultAzureCredential

    logger = logging.getLogger(__name__)
    logging.basicConfig(level=logging.INFO)


    async def backfill() -> int:
        """Returns count of rows updated."""
        endpoint = os.environ["COSMOS_ENDPOINT"]
        database_name = os.environ.get("COSMOS_DATABASE", "second-brain")
        container_name = "inbox"

        async with DefaultAzureCredential() as credential:
            async with CosmosClient(endpoint, credential=credential) as client:
                db = client.get_database_client(database_name)
                container = db.get_container_client(container_name)

                query = "SELECT * FROM c WHERE IS_DEFINED(c.foundryThreadId) AND c.foundryThreadId != null"
                count = 0
                async for item in container.query_items(
                    query=query,
                    enable_cross_partition_query=True,
                ):
                    if "foundryThreadId" not in item or item["foundryThreadId"] is None:
                        continue
                    item["sessionId"] = item["foundryThreadId"]
                    del item["foundryThreadId"]
                    await container.replace_item(
                        item=item["id"], body=item
                    )
                    count += 1
                    logger.info("Backfilled inbox %s", item["id"])

                logger.info("Backfill complete: %d rows updated", count)
                return count


    if __name__ == "__main__":
        try:
            count = asyncio.run(backfill())
            sys.exit(0)
        except Exception as exc:
            logger.exception("Backfill failed: %s", exc)
            sys.exit(1)
    ```
