    ```python
    """One-shot ADDITIVE backfill: copy Inbox.foundryThreadId → sessionId in Cosmos.

    Per Phase 24 P0-2: this script is ADDITIVE ONLY. It copies the value from
    foundryThreadId to sessionId on every doc that has foundryThreadId set
    AND does not yet have sessionId. It DOES NOT delete foundryThreadId.

    The deletion of foundryThreadId happens in plan 24-24 (post-UAT cleanup),
    only after 24-23 confirms GA UAT passed and rollback is no longer needed.

    Run from laptop BEFORE the Phase 24 deploy push (as part of pre-deploy
    gates — see 24-20 Gate 9). Idempotent — safe to re-run.

    Usage:
        cd backend && uv run python -m scripts.backfill_inbox_session_id

    Per Phase 24 task group 23.3 plan 24-17 (P0-2 amendment).
    """

    from __future__ import annotations

    import asyncio
    import json
    import logging
    import os
    import sys

    from azure.cosmos.aio import CosmosClient
    from azure.identity.aio import DefaultAzureCredential

    logger = logging.getLogger(__name__)
    logging.basicConfig(level=logging.INFO, format="%(asctime)s %(levelname)s %(message)s")


    async def backfill() -> dict[str, int]:
        """Returns counts {seen, copied, skipped_already_done, errors}."""
        endpoint = os.environ["COSMOS_ENDPOINT"]
        database_name = os.environ.get("COSMOS_DATABASE", "second-brain")
        container_name = "inbox"

        seen = 0
        copied = 0
        skipped_already_done = 0
        errors = 0

        async with DefaultAzureCredential() as credential:
            async with CosmosClient(endpoint, credential=credential) as client:
                db = client.get_database_client(database_name)
                container = db.get_container_client(container_name)

                # Match docs with foundryThreadId set, where sessionId is missing/empty
                query = (
                    "SELECT * FROM c "
                    "WHERE IS_DEFINED(c.foundryThreadId) AND c.foundryThreadId != null"
                )
                async for item in container.query_items(
                    query=query,
                    enable_cross_partition_query=True,
                ):
                    seen += 1
                    legacy = item.get("foundryThreadId")
                    existing = item.get("sessionId")
                    if not legacy:
                        # foundryThreadId is empty string or null — skip
                        skipped_already_done += 1
                        continue
                    if existing and existing == legacy:
                        # Already backfilled to the same value
                        skipped_already_done += 1
                        continue

                    try:
                        # ADDITIVE: copy without deleting foundryThreadId (P0-2)
                        item["sessionId"] = legacy
                        await container.replace_item(item=item["id"], body=item)
                        copied += 1
                        logger.info(
                            "Backfilled inbox %s (sessionId <- foundryThreadId=%s; foundryThreadId KEPT)",
                            item["id"],
                            legacy,
                        )
                    except Exception as exc:
                        errors += 1
                        logger.exception("ERROR backfilling %s: %s", item.get("id", "?"), exc)

        summary = {
            "seen": seen,
            "copied": copied,
            "skipped_already_done": skipped_already_done,
            "errors": errors,
        }
        logger.info("Backfill complete: %s", summary)
        return summary


    def main() -> int:
        try:
            summary = asyncio.run(backfill())
            print(json.dumps(summary))
            return 0 if summary["errors"] == 0 else 1
        except Exception as exc:
            logger.exception("Backfill failed: %s", exc)
            return 1


    if __name__ == "__main__":
        sys.exit(main())
    ```
