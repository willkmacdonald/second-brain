GA streaming pattern (per probe 1):
```python
from agent_framework import Agent, AgentSession, ChatOptions

session = AgentSession(session_id=stored_session_id) if stored_session_id else AgentSession()
options = ChatOptions(tool_choice="required")  # string form per probe 3

try:
    async with asyncio.timeout(60):
        async for update in agent.run(user_text, options=options, session=session, stream=True):
            for content in getattr(update, "contents", None) or []:
                if content.type == "function_call":
                    yield encode_sse({"type": "tool_call", ...})
                elif content.type == "function_result":
                    parsed = _parse_result(content.result)
                    if parsed["status"] == "filed":
                        yield encode_sse({"type": "CLASSIFIED", ...})
                    elif parsed["status"] == "low_confidence":
                        yield encode_sse({"type": "LOW_CONFIDENCE", ...})
                    elif parsed["status"] == "misunderstood":
                        yield encode_sse({"type": "MISUNDERSTOOD", ...})
                    ...
            text_delta = getattr(update, "text", "")
            if text_delta:
                yield encode_sse({"type": "text", "content": text_delta})

    final_session_id = session.session_id
    yield encode_sse({"type": "COMPLETE", "sessionId": final_session_id, "thread_id": final_session_id})

except Exception as exc:
    logger.error("Forced tool failure", exc_info=True, extra=...)
    yield encode_sse({
        "type": "ERROR",
        "sub_code": "forced_tool_failure",
        "message": "Classifier could not file this capture.",
    })
```
