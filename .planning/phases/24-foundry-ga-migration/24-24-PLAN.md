    ```python
    """Phase 24 P0-2 post-UAT cleanup: delete foundryThreadId from Inbox docs.

    Run this script ONLY after 24-23 UAT confirms GA revision is healthy and
    rollback is no longer needed. Until that point, dual-read in capture
    handler keeps backward compatibility for any rollback to the RC image.

    Idempotent — re-running finds zero rows with foundryThreadId defined.

    Usage:
        cd backend && COSMOS_ENDPOINT=https://shared-services-cosmosdb.documents.azure.com:443/ \\
            uv run python -m scripts.cleanup_foundry_thread_id

    Per Phase 24 plan 24-24 (P0-2 cleanup half).
    """

    from __future__ import annotations

    import asyncio
    import json
    import logging
    import os
    import sys

    from azure.cosmos.aio import CosmosClient
    from azure.identity.aio import DefaultAzureCredential

    logger = logging.getLogger(__name__)
    logging.basicConfig(level=logging.INFO, format="%(asctime)s %(levelname)s %(message)s")


    async def cleanup() -> dict[str, int]:
        """Returns counts {seen, cleaned, skipped_no_session_id, errors}."""
        endpoint = os.environ["COSMOS_ENDPOINT"]
        database_name = os.environ.get("COSMOS_DATABASE", "second-brain")
        container_name = "inbox"

        seen = 0
        cleaned = 0
        skipped_no_session_id = 0
        errors = 0

        async with DefaultAzureCredential() as credential:
            async with CosmosClient(endpoint, credential=credential) as client:
                db = client.get_database_client(database_name)
                container = db.get_container_client(container_name)

                query = "SELECT * FROM c WHERE IS_DEFINED(c.foundryThreadId)"
                async for item in container.query_items(
                    query=query,
                    enable_cross_partition_query=True,
                ):
                    seen += 1
                    item_id = item.get("id", "<unknown>")

                    # Safety net: don't strip foundryThreadId if sessionId is missing
                    session_id = item.get("sessionId")
                    if not session_id:
                        skipped_no_session_id += 1
                        logger.warning(
                            "SKIP %s: foundryThreadId present but sessionId missing/empty. "
                            "Run backfill_inbox_session_id.py first.",
                            item_id,
                        )
                        continue

                    try:
                        del item["foundryThreadId"]
                        await container.replace_item(item=item_id, body=item)
                        cleaned += 1
                        logger.info("Cleaned %s (sessionId=%s preserved)", item_id, session_id)
                    except Exception as exc:
                        errors += 1
                        logger.exception("ERROR cleaning %s: %s", item_id, exc)

        summary = {
            "seen": seen,
            "cleaned": cleaned,
            "skipped_no_session_id": skipped_no_session_id,
            "errors": errors,
        }
        logger.info("Cleanup complete: %s", summary)
        return summary


    def main() -> int:
        try:
            summary = asyncio.run(cleanup())
            print(json.dumps(summary))
            # Exit 1 if errors OR skipped rows exist (operator must investigate skips)
            return 0 if summary["errors"] == 0 else 1
        except Exception as exc:
            logger.exception("Cleanup failed: %s", exc)
            return 1


    if __name__ == "__main__":
        sys.exit(main())
    ```
