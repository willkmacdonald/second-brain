P0-2 dual-read helper:

```python
# backend/src/second_brain/cosmos/inbox_session_resolver.py
"""Phase 24 P0-2 dual-read helper.

During the migration window, an Inbox doc may have:
  (a) only foundryThreadId (legacy doc not yet backfilled)
  (b) both foundryThreadId AND sessionId (post-backfill, pre-cleanup)
  (c) only sessionId (new doc created after deploy + cleanup post-UAT)

resolve_inbox_session_id() prefers sessionId, falls back to foundryThreadId.
After plan 24-24 (post-UAT cleanup) deletes foundryThreadId from all docs,
the fallback path is dead but kept for defensive read of any
in-flight doc that escaped cleanup.
"""

from __future__ import annotations

from typing import Any


def resolve_inbox_session_id(inbox_doc: Any) -> str | None:
    """Return the session_id stored on this Inbox doc.

    Prefers `sessionId`. Falls back to `foundryThreadId` for docs not yet
    backfilled (P0-2 dual-read).

    Accepts any object with attribute access (Pydantic model) or
    subscriptable access (dict from Cosmos).
    """
    session_id = _read(inbox_doc, "sessionId")
    if session_id:
        return session_id
    legacy = _read(inbox_doc, "foundryThreadId")
    if legacy:
        return legacy
    return None


def _read(doc: Any, key: str) -> str | None:
    if hasattr(doc, key):
        value = getattr(doc, key, None)
        if value:
            return str(value)
    if isinstance(doc, dict) and key in doc:
        value = doc.get(key)
        if value:
            return str(value)
    return None
```
