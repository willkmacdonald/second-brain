async def session_rehydration_fresh_process() -> Path:
    """Probe 6 (Phase 24): cross-process session rehydration.

    Two-phase probe:
    - Phase A (this function in mode='turn1'): create AgentSession, send turn 1
      with magic word, persist session_id to /tmp file, exit.
    - Phase B (this function in mode='turn2'): load session_id from disk,
      reconstruct AgentSession(session_id=loaded_id), send turn 2 asking for
      the magic word, capture response text.

    Run via: `uv run python -m scripts.foundry_probe session_rehydration_fresh_process`
    The driver wrapper spawns Phase A as a subprocess, waits for exit, then spawns
    Phase B as a SEPARATE subprocess. The fresh interpreter for Phase B is the
    point — it cannot share Python-level state with Phase A.
    """
```

async def session_rehydration_fresh_process() -> Path:
    run_id = _new_run_id()
    _probe_run_id.set(run_id)
    _probe_name.set("session_rehydration_fresh_process")
    _ensure_probe_processor_installed()

    state_file = Path("/tmp") / f"foundry-probe-fresh-{run_id}.json"

    # Spawn Phase A subprocess
    a_result = subprocess.run(
        [sys.executable, "-m", "scripts.foundry_probe",
         "session_rehydration_fresh_process_phase_a", "--state-file", str(state_file)],
        capture_output=True, text=True, timeout=60,
    )

    # Read state file persisted by Phase A
    state = json.loads(state_file.read_text())
    persisted_session_id = state["session_id"]
    magic_word = state["magic_word"]

    # Spawn Phase B subprocess (fresh interpreter)
    b_result = subprocess.run(
        [sys.executable, "-m", "scripts.foundry_probe",
         "session_rehydration_fresh_process_phase_b", "--state-file", str(state_file)],
        capture_output=True, text=True, timeout=60,
    )

    state = json.loads(state_file.read_text())
    turn_two_text = state["turn_two_text"]
    recalled = magic_word.upper() in turn_two_text.upper()

    state_file.unlink(missing_ok=True)

    return _write_fixture(
        "session_rehydration_fresh_process",
        {
            "probe": {"run_id": run_id},
            "phase_a_exit_code": a_result.returncode,
            "phase_a_stderr_tail": a_result.stderr[-1000:] if a_result.stderr else "",
            "phase_b_exit_code": b_result.returncode,
            "phase_b_stderr_tail": b_result.stderr[-1000:] if b_result.stderr else "",
            "magic_word": magic_word,
            "persisted_session_id": persisted_session_id,
            "turn_two_text": turn_two_text,
            "recalled_pineapple": recalled,
        },
    )
```

    async def session_rehydration_fresh_process() -> Path:
        """Probe 6 (Phase 24, P0-1): cross-process session rehydration driver.

        Spawns Phase A and Phase B as separate subprocesses. The fresh
        interpreter for Phase B is the load-bearing test — it cannot share
        Python-level state with Phase A. Asserts that Phase B (using only the
        persisted session_id) recalls turn 1's magic word.

        Writes fixture session_rehydration_fresh_process.json with both phase
        exit codes, the magic word, the persisted session_id, the turn-2
        response text, and a boolean recalled_pineapple flag.
        """
        import subprocess
        import tempfile

        run_id = _new_run_id()
        _probe_run_id.set(run_id)
        _probe_name.set("session_rehydration_fresh_process")
        _ensure_probe_processor_installed()

        state_file = Path(tempfile.gettempdir()) / f"foundry-probe-fresh-{run_id}.json"

        # Phase A in a fresh subprocess
        a = subprocess.run(
            [
                sys.executable,
                "-m",
                "scripts.foundry_probe",
                "session_rehydration_fresh_process_phase_a",
                "--state-file",
                str(state_file),
            ],
            capture_output=True,
            text=True,
            timeout=120,
        )

        # Phase B in a SEPARATE fresh subprocess (this is the load-bearing part)
        b = subprocess.run(
            [
                sys.executable,
                "-m",
                "scripts.foundry_probe",
                "session_rehydration_fresh_process_phase_b",
                "--state-file",
                str(state_file),
            ],
            capture_output=True,
            text=True,
            timeout=120,
        )

        # Read final state from disk and compute recall
        if state_file.exists():
            state = json.loads(state_file.read_text())
            magic_word = state.get("magic_word", "")
            turn_two_text = state.get("turn_two_text", "")
            persisted_session_id = state.get("session_id", "")
            recalled = bool(magic_word) and (magic_word.upper() in turn_two_text.upper())
        else:
            magic_word = ""
            turn_two_text = ""
            persisted_session_id = ""
            recalled = False

        # Clean up state file
        state_file.unlink(missing_ok=True)

        return _write_fixture(
            "session_rehydration_fresh_process",
            {
                "probe": {"run_id": run_id},
                "phase_a_exit_code": a.returncode,
                "phase_a_stderr_tail": a.stderr[-1000:] if a.stderr else "",
                "phase_b_exit_code": b.returncode,
                "phase_b_stderr_tail": b.stderr[-1000:] if b.stderr else "",
                "magic_word": magic_word,
                "persisted_session_id": persisted_session_id,
