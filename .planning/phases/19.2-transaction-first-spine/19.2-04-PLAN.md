async function spineFetch<T>(path: string): Promise<T> {
  // Validate lazily — module-level throw breaks next build's "Collecting page data" pass.
  const BASE = process.env.SPINE_API_URL;
  const KEY = process.env.SPINE_API_KEY;
  if (!BASE || !KEY) throw new Error("SPINE_API_URL and SPINE_API_KEY env vars are required");
  const res = await fetch(`${BASE}${path}`, {
    headers: { Authorization: `Bearer ${KEY}` },
    cache: "no-store",
  });
  if (!res.ok) throw new Error(`spine ${res.status}: ${await res.text()}`);
  return (await res.json()) as T;
}
